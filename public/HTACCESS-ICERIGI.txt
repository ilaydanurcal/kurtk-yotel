# Bu dosyanın içeriğini Natro panel'den Apache Configuration'a yapıştırın
# Veya FTP ile "htaccess.txt" dosyasını ".htaccess" olarak yeniden adlandırın

# Compression (Gzip) - Google için kritik performans optimizasyonu
<IfModule mod_deflate.c>
    AddOutputFilterByType DEFLATE text/html text/plain text/xml text/css text/javascript application/javascript application/json application/xml
    AddOutputFilterByType DEFLATE image/svg+xml
</IfModule>

# Cache Control for Images and Assets
<IfModule mod_headers.c>
    # Short cache for HTML files (1 hour) for better performance
    <FilesMatch "\.(html|htm)$">
        Header set Cache-Control "public, max-age=3600, must-revalidate"
    </FilesMatch>
    
    # Cache images for 1 year but allow revalidation
    <FilesMatch "\.(jpg|jpeg|png|gif|webp|svg|ico)$">
        Header set Cache-Control "public, max-age=31536000, must-revalidate"
    </FilesMatch>
    
    # Cache CSS and JS for 1 year
    <FilesMatch "\.(css|js)$">
        Header set Cache-Control "public, max-age=31536000, must-revalidate"
    </FilesMatch>
    
    # Security headers for Google
    Header always set X-Content-Type-Options "nosniff"
    Header always set X-Frame-Options "SAMEORIGIN"
    Header always set Referrer-Policy "strict-origin-when-cross-origin"
</IfModule>

# Enable Rewrite Engine - Angular Routing için ZORUNLU
<IfModule mod_rewrite.c>
    RewriteEngine On
    
    # Handle Angular routes
    RewriteBase /
    RewriteRule ^index\.html$ - [L]
    RewriteCond %{REQUEST_FILENAME} !-f
    RewriteCond %{REQUEST_FILENAME} !-d
    RewriteRule . /index.html [L]
</IfModule>
